**Helper functions to add:**

```typescript
const SECURITY_MODES = ["ssl", "starttls", "none"] as const;

function readStr(key: string): string {
  return process.env[key] ?? "";
}

function isPresent(value: string): boolean {
  return value.trim().length > 0;
}

function requireStr(value: string, varName: string): string {
  if (!isPresent(value)) {
    console.error(`${varName} is required`);
    process.exit(1);
  }
  return value.trim();
}

function parseSecurityMode(value: string, varName: string): SecurityMode {
  const normalized = value.toLowerCase().trim();
  if (!SECURITY_MODES.includes(normalized as SecurityMode)) {
    console.error(`${varName} must be one of: ssl, starttls, none`);
    process.exit(1);
  }
  return normalized as SecurityMode;
}

function parsePort(value: string, defaultPort: number): number {
  if (!isPresent(value)) return defaultPort;
  const parsed = parseInt(value.trim(), 10);
  if (isNaN(parsed)) return defaultPort;
  return parsed;
}
```

**`loadIndexedAccount(n: number)` implementation:**
- Read all `ACCOUNT_${n}_*` env vars using `readStr()`
- `emailAddress` = `requireStr(readStr(\`ACCOUNT_${n}_EMAIL_ADDRESS\`), \`ACCOUNT_${n}_EMAIL_ADDRESS\`)`
- `imapHost` = `requireStr(readStr(\`ACCOUNT_${n}_IMAP_HOST\`), \`ACCOUNT_${n}_IMAP_HOST\`)`
- `smtpHost` = `requireStr(readStr(\`ACCOUNT_${n}_SMTP_HOST\`), \`ACCOUNT_${n}_SMTP_HOST\`)`
- `password` = `readStr(\`ACCOUNT_${n}_EMAIL_PASSWORD\`).trim()` (not required -- empty string is valid)
- `username`: if `ACCOUNT_${n}_EMAIL_USERNAME` is present use it, otherwise fall back to `emailAddress`
- `label`: if `ACCOUNT_${n}_LABEL` is present use it, otherwise `\`account${n}\``
- `imapSecurity` = parseSecurityMode of `ACCOUNT_${n}_IMAP_SECURITY`, default `"ssl"`
- `smtpSecurity` = parseSecurityMode of `ACCOUNT_${n}_SMTP_SECURITY`, default `"ssl"`
- `imapPort` = parsePort of `ACCOUNT_${n}_IMAP_PORT`, default based on imapSecurity (993 for ssl, 143 otherwise)
- `smtpPort` = parsePort of `ACCOUNT_${n}_SMTP_PORT`, default based on smtpSecurity (465 for ssl, 587 for starttls, 25 for none)
- `sslVerify` = `readStr(\`ACCOUNT_${n}_SSL_VERIFY\`) !== "false"` (matches existing pattern; note: empty string !== "false" so default is true)
- `sentFolder`: if `ACCOUNT_${n}_SENT_FOLDER` is present use it, otherwise `"Sent"`

```typescript
export function loadAccounts(): AccountConfig[] {
  const useIndexed = isPresent(readStr("ACCOUNT_1_EMAIL_ADDRESS"));
  const account1 = useIndexed ? loadIndexedAccount(1) : loadLegacyAccount();
  const accounts: AccountConfig[] = [account1];

  for (const n of [2, 3] as const) {
    const sentinel = readStr(`ACCOUNT_${n}_EMAIL_ADDRESS`);
    if (isPresent(sentinel)) {
      accounts.push(loadIndexedAccount(n));
    }
  }

  return accounts;
}
```

**`resolveAccount()` implementation:**
```typescript
export function resolveAccount(
  accounts: AccountConfig[],
  account: string | number | undefined
): AccountConfig {
  if (account === undefined) {
    return accounts[0]!;
  }

  const match = typeof account === "number"
    ? accounts.find((a) => a.label === `account${account}`)
    : accounts.find((a) => a.label.toLowerCase() === account.toLowerCase());

  if (!match) {
    const valid = accounts.map((a) => `"${a.label}"`).join(", ");
    throw new Error(`Unknown account "${account}". Valid accounts: ${valid}`);
  }

  return match;
}
```
